<!-- Swiper CSS/JS (CDN) -->
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/swiper@11/swiper-bundle.min.css">
<script src="https://cdn.jsdelivr.net/npm/swiper@11/swiper-bundle.min.js"></script>

<script>
  // 페이지에 같은 클래스의 슬라이더가 여러 개 있을 수도 있으니 querySelectorAll 사용
  document.addEventListener('DOMContentLoaded', function () {
    document.querySelectorAll('.mySwiper').forEach(function (el) {
      new Swiper(el, {
        // 기본 레이아웃
        loop: true,
        centeredSlides: true,
        slidesPerView: 1,

        // 자동 재생
        autoplay: {
          delay: 2500,        // 2.5초마다 전환
          disableOnInteraction: false // 유저가 조작해도 계속 자동재생
        },

        // 페이지 점/네비게이션 (원하면 비활성화 가능)
        pagination: {
          el: el.querySelector('.swiper-pagination'),
          clickable: true
        },
        navigation: {
          nextEl: el.querySelector('.swiper-button-next'),
          prevEl: el.querySelector('.swiper-button-prev')
        },

        // 반응형 옵션 (예: 큰 화면에서 2장씩 보기)
        // breakpoints: {
        //   1024: { slidesPerView: 2 }
        // },

        // 전환 효과 (선호에 따라 'fade', 'cube', 'coverflow' 등)
        // effect: 'fade',
        // fadeEffect: { crossFade: true },
        speed: 600 // 전환 속도(ms)
      });
    });
  });
</script>

<style>
  /* 슬라이더 내부 이미지 크기 통일 */

    width: 100%;              /* 컨테이너 폭에 맞춤 */
    object-fit: cover;        /* 비율 유지하면서 영역에 꽉 차게 */
    border-radius: 8px;       /* (선택) 모서리를 둥글게 */
  }

  /* 컨테이너 여백 등 보정 */
  .mySwiper {

  }
